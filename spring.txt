SELECT
	inv.id,
	veh.vehicleNo,
	cli.name client_name,
	cus.name customer_name,
	inv.investi_people,
	inv.investi_date,
	inv.investi_result
FROM
	bas_investigate inv
CROSS JOIN bas_vehicle veh
CROSS JOIN bas_client cli
CROSS JOIN bas_customer cus
WHERE
	inv.vehicleId = veh.id
AND veh.clientId = cli.id
AND veh.id = cus.id
AND inv.investi_date >= "2016-01-01"
